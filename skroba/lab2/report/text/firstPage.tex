\begin{itemize}
	\item \textbf{Функция 1:}
	$f(x,y) = 16x^2+ 8y^2 - 2xy + 5x - 7y + 1$\\
	\begin{equation*}
 		\begin{cases}
   			f_{x}'=0 \\
   			f_{y}'=0
 		\end{cases}
 		\Leftrightarrow minimum f(x, y)
	\end{equation*} \\
	\begin{equation*}
 		\begin{cases}
   			f_{x}'=32x - 2y + 5 = 0 \\
   			f_{y}'=16y - 2x - 7 = 0
 		\end{cases}
 		\Leftrightarrow
 		\begin{cases}
   			x = - \frac{33}{254} \\
   			y = \frac{107}{254}
 		\end{cases}
 		\Leftrightarrow
 		\min f(x,y) = - \frac{203}{254}
	\end{equation*}

   			f_{y}'=0

 		\Leftrightarrow minimum f(x, y)
	\end{equation*} \\
	\begin{equation*}

   			f_{x}'=2x - 20 = 0 \\
   			f_{y}'=6y - 12 = 0

 		\Leftrightarrow
 		\begin{cases}
   			x = 10 \\
   			y = 2
 		\end{cases}
 		\Leftrightarrow
 		\min f(x,y) = -107
